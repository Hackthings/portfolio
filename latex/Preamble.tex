%
% A chapter cover page without any following contents
%
\newcommand{\chapterpage}[3]
{
    \cleartoleftpage
    {\hfill \includegraphics[width=120pt]{#3}}

    \chapter{#1}
    {\itshape \Large #2}

    \newpage
}
%
% A chapter cover page with attached PDF contents
%
\newcommand{\chapterpagepdf}[4]
{
    \cleartoleftpage
    {\hfill \includegraphics[width=120pt]{#4}}

    \chapter{#1}
    {\itshape \Large #2}
    \newpage
    \cleartoleftpage
    \includepdf[pages={-}]{#3}
}
